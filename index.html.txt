                tab.classList.remove('bg-gray-200', 'text-gray-700');
                tab.classList.add('active', 'bg-pink-500', 'text-white');
            });
        });

        // Task completion functionality
        const tasks = document.querySelectorAll('.p-4.rounded-lg:not(.task-complete)');
        tasks.forEach(task => {
            task.addEventListener('click', function() {
                // Mark as complete
                this.classList.add('task-complete');
                
                // Update progress bar
                const progressBar = document.querySelector('.progress-bar');
                const currentWidth = parseInt(progressBar.style.width) || 40;
                progressBar.style.width = (currentWidth + 20) + '%';
                
                // Update task count
                const taskCount = document.querySelector('.text-pink-600');
                const [completed, total] = taskCount.textContent.split('/').map(Number);
                taskCount.textContent = `${completed + 1}/${total}`;
                
                // Update coins
                const coinCount = document.getElementById('coin-count');
                const reward = parseInt(this.querySelector('.bg-yellow-100').textContent.match(/\d+/)[0]);
                coinCount.textContent = parseInt(coinCount.textContent) + reward;
                
                // Add check icon
                const iconContainer = this.querySelector('.w-8.h-8');
                iconContainer.classList.remove('bg-gray-200');
                iconContainer.classList.add('bg-green-100');
                iconContainer.innerHTML = '<i class="fas fa-check text-green-500"></i>';
                
                // Change reward display
                const rewardDisplay = this.querySelector('.bg-yellow-100');
                rewardDisplay.classList.remove('bg-yellow-100', 'text-yellow-800');
                rewardDisplay.classList.add('bg-green-100', 'text-green-800');
            });
        });

        // Cat card interaction
        const feedButtons = document.querySelectorAll('.cat-card button:last-child');
        feedButtons.forEach(button => {
            button.addEventListener('click', function() {
                const hungerBar = this.closest('.cat-card').querySelector('.bg-red-400');
                const currentWidth = parseInt(hungerBar.style.width) || 0;
                hungerBar.style.width = Math.min(currentWidth + 30, 100) + '%';
                
                // Deduct coins for feeding
                const coinCount = document.getElementById('coin-count');
                const currentCoins = parseInt(coinCount.textContent);
                if (currentCoins >= 20) {
                    coinCount.textContent = currentCoins - 20;
                } else {
                    alert("Not enough coins! Complete tasks to earn more.");
                }
            });
        });

        const playButtons = document.querySelectorAll('.cat-card button:first-child');
        playButtons.forEach(button => {
            button.addEventListener('click', function() {
                const happinessBar = this.closest('.cat-card').querySelector('.bg-yellow-400');
                const currentWidth = parseInt(happinessBar.style.width) || 0;
                happinessBar.style.width = Math.min(currentWidth + 20, 100) + '%';
                
                // Increase hearts if happiness is full
                if (parseInt(happinessBar.style.width) >= 100) {
                    const heartIcons = this.closest('.cat-card').querySelectorAll('.fa-heart');
                    for (let i = 0; i < heartIcons.length; i++) {
                        if (heartIcons[i].classList.contains('text-gray-300')) {
                            heartIcons[i].classList.remove('text-gray-300');
                            heartIcons[i].classList.add('text-red-400');
                            break;
                        }
                    }
                    
                    // Update global heart count
                    const heartCount = document.getElementById('heart-count');
                    heartCount.textContent = parseInt(heartCount.textContent) + 1;
                }
            });
        });
